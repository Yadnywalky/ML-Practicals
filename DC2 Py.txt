import random
from deap import base, creator, tools, algorithms

# Define the evaluation function
# Objective: Minimize the sum of squares
def eval_func(individual):
    return sum(x ** 2 for x in individual),

# Create fitness class for minimization
creator.create("FitnessMin", base.Fitness, weights=(-1.0,))

# Create individual class
creator.create("Individual", list, fitness=creator.FitnessMin)

# Create toolbox
toolbox = base.Toolbox()

# Register attribute generator
# Random float values between -5 and 5
toolbox.register("attr_float", random.uniform, -5.0, 5.0)

# Register individual generator
# Each individual has 3 float values
toolbox.register(
    "individual",
    tools.initRepeat,
    creator.Individual,
    toolbox.attr_float,
    n=3
)

# Register population generator
toolbox.register(
    "population",
    tools.initRepeat,
    list,
    toolbox.individual
)

# Register evaluation function
toolbox.register("evaluate", eval_func)

# Register crossover operator
toolbox.register("mate", tools.cxBlend, alpha=0.5)

# Register mutation operator
toolbox.register(
    "mutate",
    tools.mutGaussian,
    mu=0,
    sigma=1,
    indpb=0.2
)

# Register selection operator
toolbox.register(
    "select",
    tools.selTournament,
    tournsize=3
)

# Create initial population
population = toolbox.population(n=50)

# Number of generations
generations = 20

# Run Genetic Algorithm
for gen in range(generations):

    # Apply crossover and mutation
    offspring = algorithms.varAnd(
        population,
        toolbox,
        cxpb=0.5,
        mutpb=0.1
    )

    # Evaluate offspring fitness
    fits = toolbox.map(toolbox.evaluate, offspring)

    # Assign fitness values
    for fit, ind in zip(fits, offspring):
        ind.fitness.values = fit

    # Select the next generation population
    population = toolbox.select(
        offspring,
        k=len(population)
    )

# Select the best individual
best_ind = tools.selBest(population, k=1)[0]

# Get best fitness value
best_fitness = best_ind.fitness.values[0]

# Print results
print("Best Individual:", best_ind)
print("Best Fitness:", best_fitness)
